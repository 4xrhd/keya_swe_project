# Database Design

## Current vs Target
- **Current implementation:** In-memory arrays (`words`, `phrases`, `languages`) + browser localStorage.
- **Target implementation:** Relational DB for multi-user persistence and scalability.

## DDL
```sql
CREATE TABLE languages (
  code            VARCHAR(10) PRIMARY KEY,
  name            VARCHAR(100) NOT NULL,
  native_name     VARCHAR(100) NOT NULL,
  direction       VARCHAR(3) NOT NULL CHECK (direction IN ('ltr','rtl')),
  flag            VARCHAR(16)
);

CREATE TABLE words (
  id              VARCHAR(20) PRIMARY KEY,
  word            VARCHAR(150) NOT NULL UNIQUE,
  part_of_speech  VARCHAR(80),
  phonetic        VARCHAR(120)
);

CREATE TABLE word_definitions (
  id              BIGSERIAL PRIMARY KEY,
  word_id         VARCHAR(20) NOT NULL REFERENCES words(id) ON DELETE CASCADE,
  sort_order      INT NOT NULL,
  definition      TEXT NOT NULL
);

CREATE TABLE word_examples (
  id              BIGSERIAL PRIMARY KEY,
  word_id         VARCHAR(20) NOT NULL REFERENCES words(id) ON DELETE CASCADE,
  sort_order      INT NOT NULL,
  example_text    TEXT NOT NULL
);

CREATE TABLE word_translations (
  id              BIGSERIAL PRIMARY KEY,
  word_id         VARCHAR(20) NOT NULL REFERENCES words(id) ON DELETE CASCADE,
  language_code   VARCHAR(10) NOT NULL REFERENCES languages(code),
  translated_text VARCHAR(200) NOT NULL,
  UNIQUE (word_id, language_code)
);

CREATE TABLE word_synonyms (
  id              BIGSERIAL PRIMARY KEY,
  word_id         VARCHAR(20) NOT NULL REFERENCES words(id) ON DELETE CASCADE,
  synonym_text    VARCHAR(150) NOT NULL
);

CREATE TABLE word_antonyms (
  id              BIGSERIAL PRIMARY KEY,
  word_id         VARCHAR(20) NOT NULL REFERENCES words(id) ON DELETE CASCADE,
  antonym_text    VARCHAR(150) NOT NULL
);

CREATE TABLE phrases (
  id              VARCHAR(20) PRIMARY KEY,
  phrase          VARCHAR(255) NOT NULL UNIQUE
);

CREATE TABLE phrase_translations (
  id              BIGSERIAL PRIMARY KEY,
  phrase_id       VARCHAR(20) NOT NULL REFERENCES phrases(id) ON DELETE CASCADE,
  language_code   VARCHAR(10) NOT NULL REFERENCES languages(code),
  translated_text VARCHAR(300) NOT NULL,
  UNIQUE (phrase_id, language_code)
);

CREATE TABLE user_profiles (
  id              UUID PRIMARY KEY,
  display_name    VARCHAR(100),
  created_at      TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE TABLE favorites (
  id              BIGSERIAL PRIMARY KEY,
  user_id         UUID NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  word_id         VARCHAR(20) NOT NULL REFERENCES words(id) ON DELETE CASCADE,
  created_at      TIMESTAMP NOT NULL DEFAULT NOW(),
  UNIQUE (user_id, word_id)
);

CREATE TABLE search_history (
  id                BIGSERIAL PRIMARY KEY,
  user_id           UUID NOT NULL REFERENCES user_profiles(id) ON DELETE CASCADE,
  query_text        VARCHAR(255) NOT NULL,
  detected_language VARCHAR(10),
  searched_at       TIMESTAMP NOT NULL DEFAULT NOW()
);
```

## Suggested Indexes
```sql
CREATE INDEX idx_words_word ON words(word);
CREATE INDEX idx_word_translations_lookup ON word_translations(language_code, translated_text);
CREATE INDEX idx_phrase_translations_lookup ON phrase_translations(language_code, translated_text);
CREATE INDEX idx_history_user_time ON search_history(user_id, searched_at DESC);
```
